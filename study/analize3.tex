\section{Papildomos laiptiniø funkcijø savybës}				% 1

\begin{ap}								% 1.1
\Def{$k$-mate gardele} (\Def{$k$-maèiu gretasieniu/staèiakampiu}) vadinama 
aibë $I \subset \R^k$, turinti pavidalà
\[ I = I_1 \times I_2 \times \cdots \times I_k
\]
kur $I_i \subset R$ -- bet kokie intervalai (áskaitant ir vientağká $[a, a] 
= \{a\}$).

$k$-matë gardelë $I$ vadinama \Def{uşdaràja}, kai visos jos ,,briaunos`` $I_i$
yra uşdaros.  Analogiğkai apibrëşiama \Def{atviroji}, \Def{aprëştoji} ir
t.t. gardelës.

Gardelës $I = \bigtimes_{n=1}^k I_n$ \Def{matu} vadinamas skaièius
\[ m(I) := \prod_{n=1}^k |I_n|
\]
(ğiuo atveju laikome, kad $0 \cdot \infty := 0$).
\end{ap}


\begin{ap}\label{ap:1.2}						% 1.2
Aibë $N \subset \R^k$ vadinama \Def{nulinio mato aibe} (şymime $m(N) = 0$),
jei $\forall \eps > 0 \quad \exists$ $k$-maèiø gardeliø seka 
$\{I_n: n \in \N\}$, su kuria
\begin{enumerate}
  \item $\displaystyle \bigcup_{n=1}^\infty I_n \supset N$ ir
  \item $\displaystyle \sum_{n=1}^\infty m(I_n) < \eps$
\end{enumerate}

\noindent Visø tokiø aibiø klasæ şymësime $\mathcal{N}$.
\end{ap}

\begin{pss}\NL								% 1.3
\begin{enumerate}
  \item Nulinio mato aibës sàvoka priklauso nuo dimensijos $k$.  Pvz.,
        tiesë $\not\in \mathcal{N}$, kai $k = 1$, bet $\in \mathcal{N}$, kai
        $k > 1$.
  \item \ref{ap:1.2} galima apsiriboti vien uşdaromis arba vien atviromis
        gardelëmis; nuo to nulinio mato aibës sàvoka nepasikeis.  

        \begin{ir*} Iğ tiesø, $m(\overline{I_n}) = m(I_n)$, tad galima 
        apsiriboti uşdaromis.  Ásitikinsime, kad galima apsiriboti ir vien
        tik atviromis gardelëmis.  Remiantis apibrëşimu laisvai pasirinkime
        $\eps > 0$.  $\exists$ gardeliø seka $\{I_n: n \in \N\}:$
        $\bigcup_{n=1}^\infty I_n \supset N$ ir $\sum_{n=1}^\infty m(I_n)
        < \eps/2$.  Kiekvienam $I_n$ galime rasti atvirà gardelæ
        $\widetilde{I}_n \subset I_n$ ir $m(\widetilde{I}_n) = 2 m(I_n)$.
        Tada $\bigcup_{n=1}^\infty \widetilde{I}_n \supset N$ ir 
        $\sum_{n=1}^\infty m(\widetilde{I}_n) < \eps$.
        \end{ir*}
\end{enumerate}
\end{pss}

\begin{tg}\label{tg:1.4}\NL						% 1.4
\begin{enumerate}
  \item $N \in \mathcal{N}$, $A \subset N$ \tada
        $A \in \mathcal{N}$.
  \item $N_i \in \mathcal{N}$, $i \in \N$ \tada
        $\displaystyle \bigcup_{i=1}^\infty N_i \in \mathcal{N}$.
        \label{tg:1.4.2}
\end{enumerate}
\end{tg}

\begin{pvzz}\NL								% 1.5
\begin{enumerate}
  \item Vientağkë aibë $\{a\}$ yra nulinio mato aibë. Remiantis \ref{tg:1.4}.%
        \ref{tg:1.4.2}, bet kuri baigtinë arba skaiti aibë taip pat yra 
        nulinio mato aibë.  Pvz., $Q \in \mathcal{N}$ ($k = 1$).
  \item $k$-matës gardelës $\bigtimes_{n=1}^k [a_n, b_n]$ ,,krağtas`` yra
        nulinio mato aibë.
        \begin{ir*}
        Iğ tikrøjø, jis sudarytas iğ $2k$ nulinio mato gardeliø $\{a_1\} \times
        [a_2, b_2] \times \cdots \times [a_k, b_k]$, $\{b_1\} \times
        [a_2, b_2] \times \cdots \times [a_k, b_k]$, \ldots, $[a_1, b_1] \times
        \cdots \times \{b_k\}$ (,,briaunø``).
        \end{ir*}
  \item Ásitikinsime, kad kiekviena glodi kreivë $\Gamma$ erdvëje $\R^k$ yra
        nulinio mato aibë ($k \ge 2$).
        \begin{ir*}
        Tarkime, $\Gamma = \gamma([a,b])$ ir $\gamma \in C^1([a,b], \R^k)$.

        Imkime bet koká intervalo $[a,b]$ skaidiná $a = t_0 < t_1 < \ldots
        < t_r = b$ ir paşymëkime $\delta := \max |t_{i+1}-t_i|$, 
        $M := \max_{t \in [a,b]} |\gamma'(t)|$.

        Remiantis Lagranşo vidutinës reikğmës teorema,
        $|\gamma(t)-\gamma(t_i)| \le M |t - t_i| \le M \delta$, kai 
        $t \in [t_i, t_{i+1}]$.

        Paşymëkime $I_i$ $k$-matæ gardelæ su centru tağke $\gamma(t_i)$ ir
        $2M|t_{i+1}-t_i|$ ilgio krağtinëmis.

        Ávertinkime suminá gardeliø $I_i$ matà:
        \begin{align*}
        \sum_{i=0}^{r-1} m(I_i) &= \sum_{i=0}^{r-1} (2M|t_{i+1}-t_i|)^k
        = (2M)^k \sum_{i=0}^{r-1} |t_{i+1}-t_i|^k
        = (2M)^k \sum_{i=0}^{r-1} \underbrace{|t_{i+1}-t_i|^{k-1}}_{\le
          \delta^{k-1}} |t_{i+1}-t_i| \\&
        \le (2M)^k \delta^{k-1} \cdot \underbrace{\sum_{i=0}^{r-1}
        |t_{i+1}-t_i|}_{= b-a} = \underbrace{(b-a)(2M)^k}_{=: C} \delta^{k-1}
        = C\delta^{k-1} \qquad \text{($C$ nepriklauso nuo skaidinio)}
        \end{align*}

        Aiğku, kad $\bigcup_{i=0}^{r-1} I_i \supset \Gamma$.

        L.p. $\eps > 0$ ir imkime bet koká $[a,b]$ skaidiná su tokiu $\delta$,
        kad $C\delta^{k-1} < \eps$.  Pagal apibrëşimà $\Gamma \in \mathcal{N}$.
        \end{ir*}
\end{enumerate}
\end{pvzz}

\begin{ap}[sàvoka ,,beveik visur``]					% 1.6
Sakoma, kad kokia nors aibës $A \subset \R^k$ tağkø savybë galioja \Def{beveik
visur} (b.v.) aibëje $A$, jei ji galioja visuose aibës $A$ tağkuose, iğskyrus
galbût nulinio mato aibæ.

\begin{description}
\item[Pvz.:]
\begin{enumerate}
  \item Funkcija $f: A \to \RR := [-\infty, \infty]$ yra baigtinë b.v., jei
        $\{ x \in A : f(x) = \pm\infty \} \in \mathcal{N}$.
  \item Funkcija $f: [a, b] \to \R$ yra b.v. tolydi, jei jos trûkio tağkø
        aibë priklauso $\mathcal{N}$.  Pvz., $\forall f \in D[a, b]$ yra
        b.v. tolydi, nes jos trûkio tağkø aibë visada baigtinë arba skaiti.
  \item Funkcijø seka $f_n \to f$ b.v., jei $\{x : f_n(x) \not\to f(x)\} \in
        \mathcal{N}$.
  \item Sakoma, kad funkcija $f: A\setminus N \to \R$ yra apibrëşta b.v.
        aibëje $A$, jei $N \in \mathcal{N}$.
\end{enumerate}
\end{description}
\end{ap}

\begin{ap}								% 1.7
Funkcija $\phi: I \to \R$ ($I \subset \R^k$ -- $k$-matë gardelë) vadinama
\Def{laiptine funkcija} jei $\exists$ baigtinë $k$-maèiø aprëştø gardeliø
sistema $\{ I_j : j = 1 \ldots m \}$, su kuria $\phi(x) = c_j$, $x \in I_j$,
$j = 1 \ldots m$ ir $\phi(x) = 0$, kai $x \in I \setminus \bigcup_{j=1}^m
I_j$.

Tokios funkcijos \Def{integralu} gardelëje $I$ vadinamas skaièius
\[
  \int_I \phi := \sum_{j=1}^m c_j m(I_j)
\]

Kiti şymëjimai:
\[
\int_I \phi(x)\,\dx
\qquad
\int_I f(x_1, \ldots, x_k)\,\dx_1\dx_2\ldots\dx_k
\]

Jei $I = [a_1, b_1] \times \cdots \times [a_k, b_k]$, dar şymima
\[
  \int_{a_1}^{b_1} \cdots \int_{a_k}^{b_k} 
    f(x_1, \ldots, x_k)\,\dx_1\dx_2\ldots\dx_k
\]

Dvimaèiu ir trimaèiu atveju dimensija $k$ parodoma integralo şenklø skaièiumi:
\[
  \iint_I f(x,y)\,\dx\dy
  \qquad
  \iiint_I f(x,y,z)\,\dx\dy\dz
\]

Visø laiptiniø funkcijø klasë gardelëje $I$ şymima $S(I)$.
\end{ap}

\begin{pss}\NL								% 1.8
\begin{enumerate}
  \item Integralo apibrëşimo korektiğkumas árodomas analogiğkai vienmaèiam
        atvejui.
  \item Laiptines funkcijas galima iğreikğti tiesinëmis gardeliø indikatoriø
        kombinacijomis: $\phi = \sum_{j=1}^m c_k \Ind_{I_j}$
        ir atvirkğèiai, visos tokio pavidalo funkcijos yra laiptinës.
\end{enumerate}
\end{pss}

\begin{tg}[laiptiniø funkcijø integralø savybës]\NL			% 1.9
\begin{enumerate}
  \item (Tiesiğkumas)
        Jei $f, g \in S(I)$, $\alpha, \beta \in \R$, tada
        \[ \int_I (\alpha f + \beta g) = \alpha \int_I f + \beta \int_I g
        \]
  \item (Monotoniğkumas)
        Jei $f, g \in S(I)$, $f \le g$, tai
        \[ \int_I f \le \int_I g
        \]
        Atskiru atveju
        \[ \left| \int_I f \right| \le \int_I |f|
        \]
\end{enumerate}
\begin{ir}
Analogiğka vienmaèiam atvejui.
\end{ir}
\end{tg}

\begin{tr}[pirmoji pagrindinë lema]					% 1.10
Tarkime, kad $\{\phi_n\} \subset S(\R^k)$ yra monotoniğkai maşëjanti
neneigiamø laiptiniø funkcijø seka, konverguojanti b.v. á $0$, kai $n \to
\infty$.  Tada
\[ \int_{\R^k} \phi_n \to 0, \quad n \to \infty
\]

\noindent
(Trumpai: jei $S(\R^k) \ni \phi_n \downarrow 0$ b.v.\footnote{monotoniğkumas
ir neneigiamumas reikalingi visur, konvergavimo uştenka ir beveik visur},
tai $\int_{\R^k} \phi_n \downarrow 0$, $n \to \infty$)
\end{tr}

\begin{isv}								% 1.11
Jei $S(\R^k) \ni \phi_n \uparrow \phi \in S(\R^k)$ (arba ,,$\downarrow$``) 
b.v., tai
\[ \int_{\R^k} \phi_n \uparrow \int_{\R^k} \phi
   \qquad \text{(atitinkamai, ,,$\downarrow$``)}
\]
\end{isv}

\begin{tr}[antroji pagrindinë lema]					% 1.12
Jei $\{ \phi_n \} \subset S(\R^k)$ -- monotoniğkai didëjanti laiptiniø 
funkcijø seka ir $A := \lim \uparrow \int_{\R^k} \phi_n < +\infty$, tai
\[ \lim_{n\to\infty} \phi_n < +\infty \text{ b.v.}
\]
\noindent (t.y., $\{ x \in \R^k : \lim_{n\to\infty} \phi_n(x) = +\infty\} \in
\mathcal{N}$).
\end{tr}

\begin{isv}								% 1.13
Jei $\{ \phi_n \} \subset S(\R^k)$, $\phi_n \ge 0$, $n \in \N$ ir
$\sum_{n=1}^\infty \int_{\R^k} \phi_n < +\infty$, tai
\[ \sum_{n=1}^\infty \phi_n < +\infty \text{ b.v.}
\]
\end{isv}


\section{Lebego integralas}						% 2

Toliau visur $I \subset \R^k$ -- bet kokia $k$-matë gardelë.

\begin{ap}								% 2.1
Laiptiniø funkcijø seka $\{ \phi_n \} \subset S(I)$ vadinama
\Def{fundamentalia} arba \Def{Koği} seka, jei
\[ \forall \eps > 0 \quad \exists N \in \N : \quad
   \int_I | \phi_n - \phi_m | < \eps
   \text{, kai $n, m > N$}
\]
\begin{ps*}
Norint atskirti tokias funkcijø Koği sekas nuo skaièiø Koği sekø kartais
sakoma, kad tokia seka yra Koği seka $L^1$ arba vidurkine pasme.
\end{ps*}
\end{ap}

\begin{ap}								% 2.2
Funkcija $f: I \to \RR$ vadinama \Def{integruojama} (Lebego prasme), jei
$\exists$ Koği seka $\{\phi_n\}\subset S(I)$, konverguojanti b.v. á $f$.
Tokiu atveju jos \Def{Lebego integralu} gardelëje $I$ vadinamas skaièius
\[ \int_I f := \lim_{n \to \infty} \int_I \phi_n
\]
\noindent Kiti integralo şymëjimai yra tokie patys kaip ir laiptiniø funkcijø.

Visø integruojamø gardelëje $I$ funkcijø klasæ şymësime $L(I)$ arba $L^1(I)$.
\end{ap}

\begin{ps}								% 2.3
Apibrëşime nurodyta riba visada egzistuoja.

\begin{ir*}
Iğ tiesø, remiantis apibrëşimu, $\forall \eps > 0$ $\exists N \in \N :$
\( \int_I |\phi_n - \phi_m| < \eps\), kai $n, m > N$, t.y.
\( |\int_I \phi_n - \int_I \phi_m | \le \int_I |\phi_n - \phi_m| < \eps\),
kai $n, m > N$.  Tai reiğkia, kad $\{ \int_I \phi_n \}$ yra skaièiø Koği seka
ir todël konverguoja.
\end{ir*}
\end{ps}

\begin{tg}[korektiğkumas]						% 2.4
Integralo apibrëşimas yra korektiğkas, t.y. jame nurodyta riba
$\lim_{n\to\infty} \int_I \phi_n$ nepriklauso nuo sekos $\{\phi_n\}$
pasirinkimo.
\end{tg}

\begin{tg}[elementarios integralo savybës]\NL				% 2.5
\begin{enumerate}
  \item Tiesiğkumas: jei $f, g \in L(I)$, $\alpha, \beta \in \R$, tada
        $\alpha f + \beta g \in L(I)$ ir
        \[ \int_I (\alpha f + \beta g) = \alpha \int_I f + \beta \int_I g
        \]
  \item Jei $f, g \in L(I)$, tai $\max \{f, g\}, \min \{f, g\}, |f| \in L(I)$
  \item Monotoniğkumas: jei $f, g \in L(I)$, $f \le g$ b.v. gardelëje $I$,
        tada
        \[ \int_I f \le \int_I g\]
        \noindent Atskiru atveju
        \[ \left| \int_I f \right| \le \int_I |f| \]
  \item Jei $f \in L(I)$, $g = f$ b.v. gardelëje $I$, tada $g \in L(I)$ ir
        \[ \int_I g = \int_I f \]
\end{enumerate}
\end{tg}

\begin{tr}[Bepo Levi teorema]						% 2.6
Jei $\{ f_n \} \subset L(I)$, $f_n \uparrow f$ ($f_n \downarrow f$) b.v.
gardelëje $I$, $\lim_{n \to \infty} \uparrow \int_I f_n < +\infty$
(atitinkamai $\lim_{n \to \infty} \downarrow \int_I f_n > -\infty$), tai $f
\in L(I)$ ir
\[ \int_I f = \lim_{n \to \infty} \int_I f_n
\]
\begin{ir}
Be árodymo.
\end{ir}
\end{tr}

\begin{isvv}\NL								% 2.7
\begin{enumerate}
  \item Jei $f_n \in L(I)$, $\sum_{n=1}^\infty \int_I |f_n| < +\infty$, tai
        $\sum_{n=1}^\infty f_n$ konverguoja absoliuèiai b.v., jos suma
        \[f := \sum_{n=1}^\infty f_n \in L(I)\qquad\text{ir}\qquad\int_I f = 
        \sum_{n=1}^\infty \int_I f_n\]
  \item Sakykime, $f \in L(I)$. Tada
        \[ \int_I |f| = 0 \ttt f = 0 \text{ b.v. gardelëje $I$}.
        \]
\end{enumerate}
\end{isvv}

\begin{tr}[Fatu--Lebego\footnotemark{} teorema]\footnotetext{Fatoux--Lebegue}\NL
									% 2.8
\begin{enumerate}
  \item[1a.] $\{ f_n \} \subset L(I)$, $f_n \le g \in L(I)$, $n \in \N$,
            $\limsup_{n\to\infty} \int_I f_n > -\infty$    \tada
            \[ \limsup_{n\to\infty} f_n \in L(I) \quad\textrm{ir}\quad
            \int_I \limsup_{n\to\infty} f_n \ge \limsup_{n\to\infty} \int_I f_n
            \]
  \item[1b.] $\{ f_n \} \subset L(I)$, $f_n \ge g \in L(I)$, $n \in \N$,
            $\liminf_{n\to\infty} \int_I f_n < +\infty$    \tada
            \[ \liminf_{n\to\infty} f_n \in L(I) \quad\textrm{ir}\quad
            \int_I \liminf_{n\to\infty} f_n \le \liminf_{n\to\infty} \int_I f_n
            \]
  \item[2.] (Lebego teorema apie maşoruotà konvergavimà)
           $\{ f_n \} \subset L(I)$, $|f_n| \le g \in L(I)$, $f_n \to f$ b.v.
           gardelëje $I$ \tada $f \in L(I)$ ir $\int_I f_n \to \int_I f$.
\end{enumerate}
\end{tr}

\begin{tg}								% 2.9
Jei $f: I \to \R$ yra aprëşta ir b.v. tolydi funkcija kompaktiğkoje gardelëje
$I$, tai $f \in L(I)$. 

\noindent
Atskiru atveju, $C(I) \subset L(I)$ kompaktiğkoje ardelëje $I$; $D[a, b]
\subset L[a, b]$.
\end{tg}

\begin{ps}								% 2.10
Galima árodyti, kad funkcijos $f$ aprëştumas ir b.v. tolydumas intervale $[a,
b]$ yra bûtina ir pakankama sàlyga, kad $f \in R[a, b]$ (integruojama Rymano
prasme). Taigi, $R[a, b] \subset L[a,b]$.
\end{ps}

\begin{ap}								% 2.11
Sakoma, kad funkcija $f: \R^k \to \RR$ yra \Def{mati}, jei $\exists$ seka
$\{ \phi_n \} \subset S(\R^k)$, su kuria $\phi_n \to f$ b.v.  Sakoma, kad
funkcija $f: A \to \RR$ ($A \subset \R^k$) yra mati, jei jos tæsinys $\hat{f}
: \R^k \to \R$, apibrëşiamas lygybe
\[ \hat{f}(x) := \begin{cases} f(x), & \text{jei $x \in A$}, \\
                               0,    & \text{jei $x \in \R^k \setminus A$}
                 \end{cases}
\]
yra mati funkcija.  Visø tokiø funkcijø aibæ şymësime $M(A)$, be to
susitarsime şymëti $M := M(\R^k)$.
\end{ap}

\begin{ap}								% 2.16

\begin{ps*}
$k$-matës gardelës ir nulinio mato aibiø atveju ğis apibrëşimas nesiskiria nuo

\end{ir*}
\end{ps*}

\begin{tg}[maèiøjø aibiø ir matø savybës]\NL				% 2.17
\begin{enumerate}
  \item Jei $A$ ir $B$ -- maèiosios aibës, tai $A \cap B$, $A \cup B$, $A
        \setminus B$, $A^\complement$ -- taip pat maèiosios aibës. Jei, be to,
        $A \cap B = \emptyset$, tai $m(A \cup B) = m(A) + m(B)$ (adityvumas), 
        o jei $A \subset B$, tai $m(A) \le m(B)$ (monotoniğkumas).
  \item Jei $A_n$, $n \in \N$ -- maèiøjø aibiø seka, tai $\bigcup_{n=1}^\infty
        A_n$ -- taip pat mati aibë.  Jei, be to, $A_i \cap A_j = \emptyset$,
        kai $i \ne j$, tai $m(\bigcup_{n=1}^\infty A_n) = \sum_{n=1}^\infty
        m(A_n)$ ($\sigma$-adityvumas).
  \item Jei $A_n$ ($n \in \N$) -- monotoniğkai didëjanti (t.y. $A_n \subset
        A_{n+1}$) maèiøjø aibiø seka, tai
        \[ m\Big(\bigcup_{n=1}^\infty A_n\Big) = \lim_{n\to\infty} \uparrow 
           m(A_n)
        \]

        Jei $A_n$ -- monotoniğkai maşëjanti maèiøjø aibiø seka ir $\exists
        n_0 \in \N:$ $m(A_{n_0}) < +\infty$, tai
        \[ m\Big(\bigcap_{n=1}^\infty A_n\Big) = \lim_{n\to\infty}
           \downarrow m(A_n)
        \]
\end{enumerate}
\begin{pss*}\NL
\begin{enumerate}
  \item Didëjanèios aibiø sekos atveju priimta rağyti $A_n \uparrow A :=
        \bigcup_{n=1}^\infty A_n$, maşëjanèios -- $A_n \downarrow A :=
        \bigcap_{n=1}^\infty A_n$.  Todël treèià savybæ galima uşrağyti taip:
        $A_n \uparrow A$ \tada $m(A_n) \uparrow m(A)$; $A_n \downarrow A$ ir
        $m(A_{n_0}) < +\infty$ \tada $m(A_n) \downarrow m(A)$.
  \item Sàlyga $m(A_{n_0}) < +\infty$ su bent vienu $n_0$ -- esminë.
        Pvz., \(A_n := [n, +\infty) \downarrow \emptyset\), bet
        $m(A_n) = +\infty \not\downarrow m(\emptyset) = 0$.
\end{enumerate}
\end{pss*}
\end{tg}

\end{tg}

\begin{ap}								% 2.19

\]
(su sàlyga, kad $f \cdot \Ind_A \in M$). Jei $\int_A f \in \R$ (t.y. $\ne
\pm\infty$), tai rağysime, kad $f \in L(A)$ ir sakysime, kad funkcija $f$ yra
\Def{integruojama} (Lebego prasme) aibëje $A$.
\end{ap}

\begin{tg}\NL								% 2.20
\begin{enumerate}
  \item (Adityvumas) $f \in L(A_1) \cap L(A_2)$, $A_1 \cap A_2 = \emptyset$
        \tada $f \in L(A_1 \cup A_2)$ ir
        \[ \int_{A_1 \cup A_2} f = \int_{A_1} f + \int_{A_2} f
        \]
  \item[1$'$.] ($\sigma$-adityvumas)
        Jei $f \in L(A)$, $A = \bigcup_{n=1}^\infty A_n$, $A_n \in M$, $n \in

  \item Tarkime, kad $A_n \uparrow A$ ($A_n \in M$, $n \in \N$). Tada

        \begin{enumerate}
          \item $f \in L(A)$
          \item $\displaystyle \lim_{n\to\infty} \int_{A_n} |f| < +\infty$

\end{enumerate}

\section[Fubinio ir kintamøjø keitimo teoremos]{Fubinio (integralo suvedimo á 
kartotinius integralus) ir kintamøjø keitimo teoremos}

\begin{tr}[Fubinio teorema]\label{tr:3.1}				% 3.1
Jei \(f \in L(I_1 \times I_2)\), \(I_1, I_2 \subset \R\) -- intervalai, tai
\[
  \int_{I_1 \times I_2} f = \int_{I_1} \Big( \int_{I_2} f(x,y)\,\dy \Big)\,\dx
                          = \int_{I_2} \Big( \int_{I_1} f(x,y)\,\dx \Big)\,\dy
\]

\begin{ps*}
Pirmoji lygybë suprantama taip: integralas \(F_1(x) := \int_{I_2}
f(x,y)\,\dy\) egzistuoja su b.v. $x \in I_1$.  Tuose tağkuose $x$, kuriuose
$F_1(x)$ neapibrëştas, susitarsime, laikyti, kad $F_1(x) := 0$.  Tuomet
$\int_{I_1} F_1(x)\,\dx = \int_{I_1 \times I_2} f$.

Analogiğkai galima komentuoti ir antràjà lygybæ.
\end{ps*}
\begin{kir}
\begin{description}
  \item[1 şingsnis:] $f = \Ind_{J_1 \times J_2}$, $J_1 \subset I_1$,
                     $J_2 \subset I_2$ -- intervalai. Tuomet
    \[
    \int_{I_1 \times I_2} f = 1 \cdot m_2(J_1 \times J_2) = m(J_1)
    \cdot m(J_2)
    \]
    ($m_2$ -- matas plokğtumoje, $m$ -- matas tiesëje).

    \begin{align*}
    \int_{I_1} \left( \int_{I_2} f(x,y)\,\dy\right)\,\dx &=
    \int_{I_1} \left( \int_{I_2} \Ind_{J_1\times J_2}(x,y)\,\dy
               \right)\,\dx
    = \int_{I_1} \left( \int_{I_2} \Ind_{J_1}(x) \Ind_{J_2}(y)\,\dy
                 \right)\,\dx \\&
    = \int_{I_1} \left( \Ind_{J_1}(x) \int_{I_2} \Ind_{J_2}(y)\,\dy
                 \right)\,\dx
    = \int_{I_1} \Ind_{J_1}(x) \cdot 1 \cdot m(J_2) \,\dx \\&
    = m(J_2) \cdot 1 \cdot m(J_1)
    = m(J_1) \cdot m(J_2)
    \end{align*}
  \item[2 şingsnis:] $f \in S(I_1 \times I_2)$

    $f$ galima iğreikğti suma
    \[ f = \sum_{i=1}^m c_i \Ind_{J_i}
      \qquad \text{$J_i$ -- staèiakampiai (dvimatës gardelës)}
    \]

    Remiantis pirmuoju şingsniu ir integralø tiesiğkumu gauname
    \begin{align*}
    \int_{I_1 \times I_2} f &= 
    \int_{I_1 \times I_2} \sum_{i=1}^m c_i \Ind_{J_i} = % tiesiğkumas
    \sum_{i=1}^m c_i \int_{I_1 \times I_2} \Ind_{J_i} =
    \sum_{i=1}^m c_i \int_{I_1} \left( \int_{I_2} \Ind_{J_i}(x,y)\,\dy
                                \right)\,\dx \\&=
    \int_{I_1} \left( \int_{I_2} \sum_{i=1}^m c_i \Ind_{J_i}(x,y)\,\dy 
               \right)\,\dx =
    \int_{I_1} \left( \int_{I_2} f(x,y) \,\dy \right)\,\dx
    \end{align*}
  \item[3 şingsnis:] $f \in L(I_1 \times I_2)$.

    Imkime laiptiniø funkcijø Koği sekà $S(I_1\times I_2) \ni f_n \to
    f$ b.v.  Tada kiekvienam $n$ turime (remdamiesi antruoju şingsniu)
    \[ \int_{I_1 \times I_2} f_n =
       \int_{I_1} \left( \int_{I_2} f_n(x,y) \,\dy \right)\,\dx
    \]
    Perëje prie ribos, kai $n \to \infty$, gauname
    \[ \int_{I_1 \times I_2} f =
       \int_{I_1} \left( \int_{I_2} f(x,y) \,\dy \right)\,\dx
    \]

    Árodymo techninis sunkumas -- grieştai pagrásti perëjima prie
    ribos değinëje lygybës pusëje.  Èia tai nebus padaryta, tad
    árodymas nepilnas.
\end{description}
\enlargethispage*{10cm}
\end{kir}
\pagebreak
\end{tr}

\begin{pss}\NL								% 3.2
\begin{enumerate}
  \item Teorema bus teisinga ir tada, kai $I_1 \subset \R^k$, $I_2 \subset
        \R^m$ -- gardelës ($k$-matë ir $m$-matë).
  \item Tarkime, $D = \{ (x, y) \in \R^2 : a \le x \le b, y_1(x) \le y \le
        y_2(x) \} \subset \R^2$, $y_1, y_2 \in M([a, b])$, $y_1 \le y_2$.

        Jei $f \in L(D)$, tai
        \begin{align*}
        \int_D f &= \iint_{\R^2} f \Ind_D
                  = \int_{-\infty}^{+\infty} \left(
                    \int_{-\infty}^{+\infty} f(x,y) \Ind_D(x,y) \,\dy
                                             \right)\,\dx
              \\ &= \int_{-\infty}^{+\infty} \left(
                    \int_{-\infty}^{+\infty} f(x,y) \Ind_{[a,b]}(x) 
                                                    \Ind_{[y_1(x),y_2(x)]}(y)
                                       \,\dy \right)\,\dx
                  = \int_{-\infty}^{+\infty} \left( \Ind_{[a,b]}(x)
                    \int_{y_1(x)}^{y_2(x)} f(x,y)\,\dy \right) \,\dx
              \\ &= \int_a^b \Big( \int_{y_1(x)}^{y_2(x)} f(x,y)\,\dy
                             \Big)\,\dx
        \end{align*}

        Analogiğkai, jei $V = \{ (x, y, z) \in \R^3: (x, y) \in D,
        z_1(x, y) \le z \le z_2(x, y) \} \subset \R^3$ ir $f \in L(V)$, tai
        \[ \int_V f = \iint_D \Big( \int_{z_1(x, y)}^{z_2(x, y)} f(x, y, z) 
           \,\dz\Big)\,\dx\dy
        \]
\end{enumerate}
\end{pss}

\begin{tr}[Tonelio teorema]						% 3.3
Jei $f \in M(I_1 \times I_2)$, $I_1, I_2$ -- intervalai, $f \ge 0$, tai
\begin{equation}
   \int_{I_1 \times I_2} f = \int_{I_1} \Big( \int_{I_2} f(x,y)\,\dy \Big)
   \,\dx = \int_{I_2} \Big( \int_{I_1} f(x,y)\,\dx \Big) \,\dy
   \tag*{$(\le +\infty)$}
\end{equation}

Èia galioja ta pati pastaba kaip ir Fubinio teoremai (\ref{tr:3.1}).  Ği
teorema tinka visoms kvaziintegruojamoms funkcijoms.
\end{tr}

\begin{isv}								% 3.4
Jei $f \in M(I_1 \times I_2)$ ir $\int_{I_1} \big( \int_{I_2} |f(x,y)|\,\dy
\big)\,\dx < +\infty$, tai $f \in L(I_1 \times I_2)$ ir tuo paèiu funkcijai
$f$ teisinga Fubinio teorema (\ref{tr:3.1}).
\end{isv}

\begin{pvzz}\NL								% 3.5
\begin{enumerate}
  \item Tarkime, $f \in L(I_1)$, $g \in L(I_2)$. Tada
        \[ \int_{I_1 \times I_2} f \circledast g = \int_{I_1} f \cdot 
           \int_{I_2} g
        \]
        ($f \circledast g (x,y) := f(x) \cdot g(y)$ -- \Def{tenzorinë
        sandauga}).
        \begin{ir*}
        \[
        \iint_{I_1 \times I_2} f(x)g(y)\,\dx\dy 
         = \int_{I_1} \left( f(x) \int_{I_2} g(y)\,\dy \right)\,\dx
         = \int_{I_1} f(x)\,\dx \cdot \int_{I_2} g(y) \,\dy
        \]
        \end{ir*}

        Pvz:
        \[
        \iint_{\R^2} \frac{\dx\dy}{(1+x^2)(1+y^2)} 
        = \left(\int_{-\infty}^{+\infty} \frac{\dx}{1+x^2}\right)^2 = \pi^2
        \]
  \item Tarkime, $\displaystyle f(x, y) := \frac{x^2 - y^2}{(x^2 + y^2)^2}$
        \begin{align*}
          \int_0^1 \dx \int_0^1 f(x,y)\,\dy &= \frac{\pi}{4} \\
          \int_0^1 \dy \int_0^1 f(x,y)\,\dx &= -\frac{\pi}{4}
        \end{align*}
        Gauname skirtingus rezultatus, nes $f \not\in L([0,1]^2)$.
\end{enumerate}
\end{pvzz}

\medskip
Toliau $D$ ir $\widetilde{D}$ -- bet kokios atviros aibës erdvëje $\R^k$.

\begin{ap}								% 3.6
Diferencijuojamos funkcijos $f: D \to \R^k$ \Def{Jakobianu} tağke $x \in D$
vadinamas skaièius
\[
  J_f(x) := \frac{D(f_1, f_2, \ldots, f_k)}{D(x_1, x_2, \ldots, x_k)} :=
            \det f'(x) = \begin{vmatrix}
              \frac{\partial{f_1}}{\partial{x_1}} &
                \frac{\partial{f_1}}{\partial{x_2}} &
                  \ldots &
                    \frac{\partial{f_1}}{\partial{x_k}} \\
              \frac{\partial{f_2}}{\partial{x_1}} &
                \frac{\partial{f_2}}{\partial{x_2}} &
                  \ldots &
                    \frac{\partial{f_2}}{\partial{x_k}} \\
              \vdots &
                \vdots &
                  \ddots &
                    \vdots \\
              \frac{\partial{f_k}}{\partial{x_1}} &
                \frac{\partial{f_k}}{\partial{x_2}} &
                  \ldots &
                    \frac{\partial{f_k}}{\partial{x_k}} \end{vmatrix}
\]
\end{ap}

\begin{tr}[kintamøjø keitimo teorema]					% 3.7
Jei $g \in C^1(D, \widetilde{D})$ yra bijekcija, $\forall x \in D$ $J_g(x) 
\ne 0$ ir $f \in L(\widetilde{D})$, tai
\[ \int_{\widetilde{D}} f(y)\,\dy = \int_D f\big(g(x)\big) \, |J_g(x)| \,\dx
\]
arba
\[ \int_{\widetilde{D}} f(y_1, \ldots, y_k)\,\dy_1\dots\dy_k = 
   \int_D f\big(g_1(x_1, \ldots, x_k), \ldots, g_k(x_1, \ldots, x_k)\big) \, 
   \left|\frac{D(g_1, \ldots, g_k)}{D(x_1, \ldots, x_k)}\right| 
   \,\dx_1 \dots \dx_k
\]
(èia $g = (g_1, \ldots, g_k)$).

\begin{ir}
Be árodymo.
\end{ir}
\end{tr}

\begin{pvzz}\NL								% 3.8
\begin{enumerate}
  \item (Polinis kintamøjø keitimas)

        Imkime \(\widetilde{D} = \R^2 \ni (x, y) \longleftrightarrow (r, \phi)
        \in D = [0; +\infty) \times [0; 2\pi]\)
        \[ \begin{cases} x = r \cos \phi \\ y = r \sin \phi \end{cases}
        \]

        Kadangi nëra vienareikğmiğkumo, galime imti \(\widetilde{D}' :=
        \widetilde{D} \setminus [0; +\infty) \times \{0\}\) ir \(D' := (0;
        +\infty) \times (0; 2\pi)\). Integralas nuo to nepasikeis.

        \[ \iint_{\R^2} f(x, y)\,\dx\dy = \int_0^{2\pi} \dphi
           \int_0^{\infty} f(r\cos\phi, r\sin\phi) \, \left|\frac{D(x, y)}%
           {D(r, \phi)}\right|\,\dr
        \]

        Jakobianas
        \[ \frac{D(x, y)}{D(r, \phi)} =
        \begin{vmatrix}
             \frac{\partial{x}}{\partial{r}} &
               \frac{\partial{x}}{\partial{\phi}} \\
             \frac{\partial{y}}{\partial{r}} &
               \frac{\partial{y}}{\partial{\phi}}
           \end{vmatrix} =
           \begin{vmatrix}
             \cos\phi & -r\sin\phi \\
             \sin\phi & -r\cos\phi \\
           \end{vmatrix} =
           r (\cos^2 \phi + \sin^2 \phi) = r
        \]

        Pvz.:
        \[ \iint_{x^2+y^2 \le R^2} f(x, y) \,\dx\dy =
           \int_0^{2\pi} \dphi \int_0^R f(r\cos\phi, r\sin\phi)\,r\,\dr
        \]

        Kitas pvz.:
        \[ I = \iint_{\R^2} e^{-(x^2+y^2)}\,dx\dy
             = \int_0^{2\pi} \dphi \int_0^{+\infty} e^{-r^2}\,
               \underbrace{r\dr}_{= \frac{1}{2}\dr^2}
             = 2\pi \cdot \left(-\frac{e^{-r^2}}{2}\right)\Bigg|_0^{+\infty}
             = \pi
        \]
        Beje,
        \[ I = \left(\int_{-\infty}^{+\infty} e^{-x^2}\,\dx\right)^2 = \pi
           \tada
           \int_{-\infty}^{+\infty} e^{-x^2}\,\dx = \sqrt{\pi}
        \]
  \item (Sferinës koordinatës)
        \[ \begin{cases} x = r \sin \theta \cos \phi \\
                         y = r \sin \theta \sin \phi \\
                         z = r \cos \theta \end{cases}
        \]
        Jei $(x, y) \ne (0, 0)$, tai $(x, y, z)$ atitinka vienintëlis trejetas
        $(r, \phi, \theta)$, kai $r \in (0; +\infty)$, \(\phi \in [0; 2\pi)\),
        $\theta \in (0; \pi)$ (kartais imama \(\theta \in (-\frac{\pi}{2};
        \frac{\pi}{2})\)).

        Jakobianas

\begin{pvzz}[plotø ir tûriø skaièiavimas]\NL				% 4.1
\begin{enumerate}
  \item Aibës (,,figûros``) $A \subset \R^2$ \Def{plotu} vadinamas jos Lebego
        matas $m(A)$ (su sàlyga, kad $A$ -- mati aibë).

        Imkime $A = \{(x, y) : a \le x \le b, y_1(x) \le y \le y_2(x) \}$,
        $y_1 \le y_2$ -- maèios funkcijos (t.y., $A$ -- kreivinë trapecija).
        \begin{align*}
          m(A) &= \iint_A\dx\dy 
                = \iint_{R^2} \Ind_A(x,y)\,\dx\du 
                = \iint_{R^2} \Ind_{[a;b]}(x)\Ind_{[y_1(x);y_2(x)]}(y)\,\dx\dy
            \\ &= \int_{-\infty}^{+\infty} \Big(\int_{-\infty}^{+\infty} 
                       \Ind_{[a;b]}(x)\Ind_{[y_1(x);y_2(x)]}(y)\,\dy\Big)\,\dx
            \\ &= \int_a^b \Big(\int_{y_1(x)}^{y_2(x)} \dy\Big)\,\dx
                = \int_a^b \big(y_2(x)-y_1(x)\big)\,\dx
        \end{align*}

  \item Analogiğkai aibës (,,kûno``) $A \subset \R^3$ \Def{tûriu} vadinamas jo
        Lebego matas $m(A)$.

        Jei $A = \{(x,y,z): (x,y) \in D, z_1(x, y) \le z \le z_2(x, y) \}$,
        $D \subset \R^2$ ($A$ -- kreivinis cilindras), tai
        \[ m(A) = \iint_D \big(z_2(x,y) - z_1(x,y)\big)\,\dx\dy
        \]
        (su sàlyga, kad $D$ -- mati aibë ir $z_1 \le z_2$ -- maèios funkcijos
        aibëje $D$).

  \item Aibës $A \subset \R^3$ \Def{pjûviu} tağke $x \in \R$ vadinama aibë
        $A_x := \{(y,z): (x,y,z) \in A\}$.

        Tada tûris yra lygus
        \begin{align*}
        m(A) &= \iiint_{\R^3} \Ind_A(x,y,z)\,\dx\dy\dz
              = \iiint_{\R^3} \Ind_{A_x}(y,z)\,\dx\dy\dz
          \\ &= \int_{-\infty}^{+\infty} \Big( \iint_{\R^2} \Ind_{A_x} (y,z)
                \,\dy\dz \Big)\,\dx
              = \int_{-\infty}^{+\infty} S(A_x)\,\dx
        \end{align*}
        èia $S(A_x) = m_2(A_x)$ -- aibës $A_x$ plotas.
\end{enumerate}

\begin{pvz}[pavirğiaus ploto skaièiavimai]				% 4.2
Imkime pavirğiø $\Phi = \{ (x,y,z) : (x,y)\in D, z = f(x,y) \}$,
$f: D\to\R$ -- ,,gera`` funkcija.
Kam lygus pavirğiaus $\Phi$ plotas?

Padalinkime $D$ á daug maşø sriteliø:
\[ D = \bigcup_{i=1}^m D_i \qquad D_i \cap D_j = \emptyset,\text{ kai $i\ne j$}
\]

Kiekvienà $D_i$ atitinka pavirğiaus dalis $\Phi_i = \{(x,y,z): (x,y) \in D_i,
z = f(x,y) \}$.  Kiekvienam $\Phi_i$ imkime tağkà $(x_i, y_i, z_i) \in
\Phi_i$.  Iğveskime per já lieèiamàjà plokğtumà ir paşymëkime
$\widetilde{\Phi}_i$ tos plokğtumos dalá, esanèià virğ srities $D_i$.

Intuityviai aiğku, kad
\[ S(\Phi_i) \approx S(\widetilde{\Phi}_i) = \frac{S(D_i)}{|\cos \alpha_i|}
\]
kur $\alpha_i$ -- kampas tarp $\widetilde{\Phi}_i$ ir plokğtumos $Oxy$.

Iğveskime $\widetilde{\Phi}_i$ lygtá:
\[ f(x,y) = f(x_i, y_i) + \frac{\partial{f}}{\partial{x}}(x_i,y_i)(x-x_i)
                        + \frac{\partial{f}}{\partial{y}}(x_i,y_i)(y-y_i)
                        + o(\sqrt{(x-x_i)^2+(y-y_i)^2})
\]
kai $x \to x_i$, $y \to y_i$.

Atmetæ ,,maşà`` nará $o(\sqrt{(x-x_i)^2+(y-y_i)^2})$, gauname plokğtumà
\[ z = \underbrace{f(x_i, y_i)}_{= z_i} + 
        \frac{\partial{f}}{\partial{x}}(x_i,y_i)(x-x_i) +
        \frac{\partial{f}}{\partial{y}}(x_i,y_i)(y-y_i)
\]

Ği plokğtuma ir vadinama \Def{pavirğiaus lieèiamàja plokğtuma} tağke $(x_i,
y_i, z_i)$.

Primename kampo tarp plokğtumø $ax+by+cz+d=0$ ir $\tilde{a}x+\tilde{b}y+
\tilde{c}z+\tilde{d}=0$ formulæ:
\[ \cos\alpha = \frac{|a\tilde{a}+b\tilde{b}+c\tilde{c}|}
                {\sqrt{a^2+b^2+c^2}\sqrt{\tilde{a}^2+\tilde{b}^2+\tilde{c}^2}}
\]


\newcommand{\BjauriSaknis}[1][_i]
           {\sqrt{1 + \left(\frac{\partial{f}}{\partial{x}}(x#1,y#1)\right)^2
                    + \left(\frac{\partial{f}}{\partial{y}}(x#1,y#1)\right)^2}}

Taigi, kampas $\alpha_i$ yra
\[ \cos\alpha_i = \frac{1}{\BjauriSaknis}
\]

Ástatæ já á $S(\widetilde{\Phi}_i)$ iğraiğkà gauname
\[ S(\widetilde{\Phi}_i) = S(D_i) \cdot \BjauriSaknis
\]

Tada
\[
S(\Phi) = \sum_i S(\Phi_i) \approx \sum_i S(\widetilde{\Phi}_i) 
        = \sum_i \BjauriSaknis S(D_i)
\]

Paşymëkime
\( \phi(x, y) := \BjauriSaknis \), $(x, y) \in D_i$,
t.y.
\[ \phi := \sum_i \BjauriSaknis \Ind_{D_i}
\]

Tuomet
\begin{align*}
S(\Phi) &\approx \sum_i \BjauriSaknis S(D_i) \\
        &=        \sum_i \BjauriSaknis \iint_D \Ind_{D_i} \\
        &= \iint_D \sum_i \BjauriSaknis \Ind_{D_i} \\
        &= \iint_D \phi
\end{align*}

Funkcija $\phi$ priklauso nuo srities $D$ padalinimo á sriteles $D_i$ ir tağkø
$(x_i, y_i) \in D_i$ pasirinkimo.  $\phi$ artëja prie $\psi$:
\[ \psi(x,y) := \BjauriSaknis[]
\]
su sàlyga, kad $\frac{\partial f}{\partial x}$ ir $\frac{\partial f}{\partial
y}$ -- tolydşios funkcijos.

Taigi,
\[ S(\Phi) = \iint_D \BjauriSaknis[]\,\dx\dy
\]
arba sutrumpintai
\[ S(\Phi) = \iint_D \sqrt{1+(z'_x)^2+(z'_y)^2}\,\dx\dy
\]

Pvz., imkime $z := \sqrt{R^2-x^2-y^2}$, $(x,y) \in D := \{(x,y): x^2+y^2 \le
R^2\}$
\begin{align*}

                    +\left(\frac{-2y}{2\sqrt{R^2-x^2-y^2}}\right)^2}\,\dx\dy \\
    &= 2 \iint_D \sqrt{1 + \frac{x^2+y^2}{R^2-x^2-y^2}}\,\dx\dy 
     = 2 \iint_D \sqrt{\frac{R^2}{R^2-x^2-y^2}}\,\dx\dy \\
    &= 2R \iint_D \frac{\dx\dy}{\sqrt{R^2-x^2-y^2}}

              \Bigg|_0^R

\end{align*}

\begin{pvz}[kûno masës skaièiavimas]					% 4.3
Tarkime, kad ,,kûnas`` $A \subset \R^3$ kiekvienam savo tağke $(x,y,z)$ turi
\Def{tanká} $\rho(x,y,z) \ge 0$ ir galima $A$ padalinti á atskirus gabalus, 
kuriuose tankis $\rho$ yra pastovus:
\[
  \rho(x,y,z) = \rho_i \qquad x,y,z \in A_i
                       \qquad A = \bigcup_i A_i
                       \qquad A_i \cap A_j = \emptyset,\text{ kai $i \ne j$}
\]

Tada kûno $A$ \Def{masë} lygi
\begin{align*}
  M(A) &= \sum_i \rho_i \underbrace{V(A_i)}_{\text{tûris}}
        = \sum_i \rho_i \iiint_{\R^3} \Ind_{A_i}
        = \iiint_{\R^3} \left(\sum_i \rho_i \Ind_{A_i} \right)
    \\ &= \iiint_A \rho(x,y,z)\,\dx\dy\dz
\end{align*}

Ğia lygybe apibrëğime \Def{masæ} bet kokiam kûnui:
\[ M(A) = \iiint_A \rho(x,y,z)\,\dx\dy\dz
   \qquad \rho  \ge 0 \text{ -- bet kokia mati funkcija}
\]
\end{pvz}

\begin{pvz}[materialaus kûno statinis momentas ir masës centras]	% 4.4
Materialaus tağko $a = (x,y,z)$ su mase $m$ \Def{statinis momentas} $Oxy$
plokğtumos atşvilgiu -- tai skaièius $z \cdot m$ (atstumo iki plokğtumos ir
masës sandauga).

Jei turime keliø materialiø tağkø sistemà, tai jos statinis momentas
apibrëşiamas kaip atskirtø tağkø statiniø momentø suma.

Turime materialø kûnà $A \subset \R^3$, kurio tankis tağke $(x,y,z)$ yra
$\rho(x,y,z) \ge 0$.

Tarkime, kad $A = \bigcup_{i=1}^m A_i$, $A_i \cap A_j = \emptyset$, kai
$i \ne j$, ir $\rho(x,y,z) = \rho_i$, kai $(x,y,z) \in A_i$.

Tada kûno $A$ statinis momentas
\[ M_{xy}(A) \approx \sum_{i=1}^m z_i \underbrace{\rho_i V(A_i)}_{= M(A_i)}
\qquad\text{(èia $(x_i, y_i, z_i)$ -- bet koks tağkas iğ $A_i$)}
\]

Tæsdami galime uşrağyti
\begin{align*}
M_{xy}(A) &\approx \sum_i z_i \rho_i \iiint_{\R^3} \Ind_{A_i}
          = \iiint_{\R^3} \left( \sum_i \rho_i z_i \Ind_{A_i} \right)
          \approx \iiint_{\R^3} z \underbrace{\sum_i \rho_i \Ind_{A_i}}_{= 
                                                                       \rho}
      \\ &= \iiint_A z \cdot \rho(x,y,z)\,\dx\dy\dz
\end{align*}

Natûralu ğià formulæ laikyti statinio momento apibrëşimu bet kokiam kûnui.
Taip pat galime apibrëşti statinius momentus ir kitoms koordinaèiø
plokğtumoms:
\begin{align*}
  M_{xy}(A) &= \iiint_A z \cdot \rho(x,y,z)\,\dx\dy\dz \\
  M_{xz}(A) &= \iiint_A y \cdot \rho(x,y,z)\,\dx\dy\dz \\
  M_{yz}(A) &= \iiint_A x \cdot \rho(x,y,z)\,\dx\dy\dz
\end{align*}

Kûno $A$ \Def{masës centru} vadinamas tağkas $(x_0, y_0, z_0)$, kuriame
sukoncentravus visà kûno masæ, statiniai momentai nepasikeistø.%
\footnote{Galima árodyti, kad jei statiniai momentai iğliks pastovûs
$Oxy$, $Oxz$ ir $Oyz$ plokğtumø atşvilgiu, tai jie iğliks pastovûs ir visø kitø
plokğtumø atşvilgiu.}

Masës centro koordinatës yra tokios:
\[
  x_0 = \frac{M_{yz}(A)}{M(A)} \qquad
  y_0 = \frac{M_{xz}(A)}{M(A)} \qquad
  z_0 = \frac{M_{xy}(A)}{M(A)}
\]
\end{pvz}

\section{Integralai, priklausantys nuo parametro}

Daşnai tenka nagrinëti tokio pavidalo funkcijas:
\[ J(x) := \int_Y f(x, y) \,\dy.
\]
Jos vadinamos integralais, priklausanèiais nuo parametro (sutrumpintai IPP).

Toliau laikysime, kad $X \subset \R^k$, $Y \subset \R^n$ -- maèios aibës, $f:
X \times Y \to \R$ tokia, kad $\forall x \in X \quad f(x, \cdot) \in L(Y)$
(èia ir kitur, $f(x, \cdot)$ şymi funkcijà $g_x(y) : Y \to \R$, $\forall y \in
Y \quad g_x(y) := f(x, y)$).

\begin{ap}								% 5.1
Sakysime, kad IPP $J(x) := \int_Y f(x,y)\,\dy$ yra \Def{normaliai
integruojamas} aibëje $A \subset X$ (arba ,,$x \in A$ atşvilgiu``, arba
,,pagal $x \in A$``), jei $\exists g \in L(Y):$
\[ |f(x,y)| \le g(y), \qquad y \in Y, \quad x \in A
\]
(kartais dar sakoma ,,integralas normaliai konverguoja``).
\end{ap}

\begin{tg}								% 5.2
Tarkime, kad funkcija $f: X \times Y \to \R$ tenkina ğias sàlygas:
\begin{enumerate}
\item $\forall y \in Y \quad \exists f_0(y) := \lim_{x \to x^0} f(x,y)$.
\item $\exists$ pradurtoji tağko $x^0$ aplinka $\dot{U}$ tokia, kad IPP $J$
yra normaliai integruojamas aibëje $\dot{U} \cap X$ (t.y. $\exists g \in L(Y)
\quad |f(x,y)| \le g(y) \quad y \in Y, \quad x \in \dot{U} \cap X$).
\end{enumerate}
Tada $\exists J_0 := \lim_{x \to x^0} J(x) := \int_Y f_0(y) \,\dy$
(t.y. \[\lim_{x \to x^0} \int_Y f(x,y)\,\dy = \int_Y \lim_{x\to x^0}
f(x,y)\,\dy ).\]
Atskiru atveju, kai vietoje pirmos sàlygos iğplildyta tokia:
\begin{enumerate}
\item[1$'$.] $\forall y \in Y$ funkcija $f(\cdot, y)$ yra tolydi tağke
             $x^0$ (t.y. $f_0(y) = f(x^0,y)$),
\end{enumerate}
$J$ yra tolydus tağke $x^0$.
\end{tg}

\begin{pvzz}\NL								% 5.3
\begin{enumerate}
\item Funkcijos \(f \in L[0; +\infty)\) \Def{Laplaso transformacija} vadinama
      funkcija \[ F(p) := \int_0^{+\infty} e^{-px}f(x)\,\dx, \qquad p \ge 0 \]

      Du faktai:
      \begin{enumerate}
        \item pointegralinë funkcija $h(p, x) = e^{-px} f(x)$ yra tolydi $p$
              atşvilgiu;
        \item $|h(p,x)| \le |f(x)|$, $p \ge 0$, $x \ge 0$.
      \end{enumerate}

      Kadangi \(|f| \in L[0; +\infty)\), tai IPP $F(p)$ normaliai
      integruojamas visame intervale \([0; +\infty)\).  Remdamiesi praeita
      teorema gauname, kad Laplaso transformacija visada yra toldydi funkcija.
\item Tarkime, $f \in L(\R)$.  Kompleksinë funkcija \[\hat{f}(t) :=
      \int_{-\infty}^{+\infty} e^{itx} f(x) \,\dx, \qquad t \in \R,\] vadinama
      funkcijos $f$ \Def{Furje transformacija}.

      (Primename, kad \(e^{ix} := \cos x + i \sin x\); \(\int(f(x) + i
      g(x))\,\dx := \int f(x)\,\dx + i \int g(x)\,\dx\), \(f, g: \R \to \R\))

      Faktai:
      \begin{enumerate}
        \item IPP tolydus $t$ atşvilgiu;
        \item $|e^{itx} f(x)| = |f(x)|$.
      \end{enumerate}

      Taigi Furje transformacija visada yra tolydi.
\item Tarkime, kad $f \in C(X \times Y)$ ir $X, Y$ -- kompaktai.  Tada
      IPP $J(x) := \int_Y f(x,y) \,\dy$, $x \in X$ -- tolydi funkcija.
\item $F(x) := \int_0^{+\infty} \underbrace{xe^{-xy}}_{=: f(x,y)}\,\dy$, $x
      \ge 0$, $f(x,y)$ tolydi pagal $x$.

      \begin{align*}
      F(0) &= 0 \\
      F(x) &= -e^{-xy} \Big|_{y=0}^{+\infty} = 1 \quad x \ge 0
      \end{align*}

      Taigi, $F(x) \not\in C$ (nëra tolydus tağke $x = 0$).
\end{enumerate}
\end{pvzz}

\begin{tr}								% 5.4
Tarkime, kad funkcija $f: X \times Y \to \R$ tenkina ğias sàlygas:
\begin{enumerate}
\item $\forall y \in Y \quad \exists \frac{\partial f(x^0,y)}{\partial x_i}$
      (pagal kaşkokià vienà koordinatæ $x_i$).
\item $\exists$ tağko $x^0$ aplinka $U$, kurioje funkcijø sistema

\begin{ap}[su prisiminimu]						% 7.1
\label{7.1}
\Def{Kreive} erdvëje $\R^k$ vadinama bet kokia funkcija $\gamma \in C([a,b],
\R^k)$.  Kreivë $\gamma$ vadinama \Def{glodşia}, jei ji yra tolydşiai
diferencijuojama ir $\gamma'(t) \ne 0$ (kai kada pastaroji sàlyga á apibrëşimà
neátraukiama).  Kreivë $\gamma$ vadinama \Def{gabalais} (\Def{dalimis})
\Def{glodşia}, jei $\gamma'(t)$ egzistuoja ir yra aprëşta iğskyrus galbût
baigtiná\footnote{Galima bûtø kalbëti apie skaièià aibæ, bet taikymuose tai
retai pasitaiko.} skaièiø tağkø.

Funkcija $\tau : [a,b] \to [\alpha, \beta]$ vadinama \Def{diffeomorfizmu}, jei
ji yra tolydşiai diferencijuojama bijekcija, kurios atvirkğtinë $\tau^{-1}$
taip pat yra tolydşiai diferencijuojama.

\shortpage
Sakoma, kad kreivës $\gamma_1$ ir $\gamma_2$ yra \Def{ekvivalenèios}, jei
egzistuoja grieştai didëjantis diffeomorfizmas $\tau$ toks, kad $\gamma_1 =
\gamma_2 \circ \tau$ (t.y. $\gamma_1(t) = \gamma_2(\tau(t))$ su visais $t$).
Tokiu atveju rağoma $\gamma_1 \sim \gamma_2$ ir sakoma, kad kreivë $\gamma_1$
yra \Def{gaunama} iğ kreivës $\gamma_2$ parametro keitimu $\tau$.

Glodşios kreivës $\gamma : [a,b] \to \R^k$ \Def{liestine} tağke $t_0 \in
[a,b]$ vadinama tiesë
\[ x(t) = \gamma(t_0) + \gamma'(t_0) (t-t_0), \qquad t \in \R.
\]

Vektorius $\gamma'(t_0)$ yra \Def{lieèiamasis vektorius} kreivei $\gamma$
tağke $t_0$ (geometriğkai -- tağke $\gamma(t_0)$).

Vektorius $T(t_0) := \frac{\gamma'(t_0)}{|\gamma'(t_0)|}$ vadinamas kreivës
$\gamma$ \Def{vienetiniu lieèiamuoju vektoriumi}.
\end{ap}

\begin{tg}[kreiviø ekvivalentumo savybës]\NL				% 7.2
\begin{enumerate}
\item \label{7.2.1} $\gamma \sim \gamma$.
\item \label{7.2.2} $\gamma_1 \sim \gamma_2$ \tada $\gamma_2 \sim \gamma_1$.
\item \label{7.2.3} $\gamma_1 \sim \gamma_2$ ir $\gamma_2 \sim \gamma_3$ \tada
      $\gamma_1 \sim \gamma_3$.
\item Vienetinis lieèiamasis vektorius nesikeièia keièiant parametrà.
\end{enumerate}
\begin{ps*} \ref{7.2.1}--\ref{7.2.3} savybës leidşia visas kreives erdvëje
$\R^k$ suskirstyti á nesikertanèias klases.  Kiekvienà tokià klasæ galima
interpretuoti kaip vienà geometrinæ kreivæ su pasirinkta ,,apëjimo kryptimi``.
\end{ps*}
\end{tg}

\begin{ap}								% 7.3
\Def{Orientuota} kreive (erdvëje $\R^k$) vadinama bet kokia klasë ekvivalenèiø
kreiviø erdvëje $\R^k$.

Kreivë $\overline{\gamma}(t) := \gamma(a+b-t)$, $t \in [a,b]$, vadinama
kreive, gauta iğ kreivës $\gamma : [a, b] \to \R^k$ \Def{orientacijos}

\begin{pss}\NL								% 7.4
\begin{enumerate}

\gamma_1 \sim \gamma_2 &\tada \int_{\gamma_1} \langle f, \dx \rangle =
                              \int_{\gamma_2} \langle f, \dx \rangle; \\
\gamma_1 \sim \overline{\gamma_2} &\tada \int_{\gamma_1} \langle f, \dx \rangle
                                      = -\int_{\gamma_2} \langle f, \dx \rangle.
\end{align*}

Ğios savybës leidşia aibrëşti antro tipo kreiviná integralà geometrine kreive
$\Gamma \subset \R^k$ su pasirinkta ,,apëjimo`` kryptimi:
\[ \int_\Gamma \langle f, \dx \rangle := \int_\gamma \langle f, \dx \rangle,
\]
kur $\gamma$ -- $\Gamma$ parametrizacija su ,,atitinkamai`` pasirinkta
kryptimi.  Pakeitus ,,apëjimo kryptá`` prieğinga, integralo reikğmë keis
şenklà.

\item (Tiesiğkumas)
\[ \int_\gamma \langle af+bg, \dx \rangle = a \int_\gamma \langle f, \dx
   \rangle + b \int_\gamma \langle g, \dx \rangle.
\]

\item (Adityvumas)
Jei $\gamma = \gamma_1 \cup \gamma_2$, tai
\[ \int_\gamma \langle f, \dx \rangle = \int_{\gamma_1} \langle f, \dx
   \rangle + \int_{\gamma_2} \langle f, \dx \rangle.
\]

\item
\[ \Big| \int_\gamma \langle f, \dx \rangle \Big| \le \int_\gamma |f|\,\dl.
\]

\end{enumerate}
\end{tgg}

\begin{ap}								% 7.12
Aibë $D \subset \R^k$ vadinama \Def{lankiğkai jungia}, jei $\forall x, y \in
D$ egzistuoja gabalais glodi kreivë $\gamma : [\alpha, \beta] \to D$ su galais
$\gamma(\alpha) = x$, $\gamma(\beta) = y$.

\Def{Sritimi} erdvëje $\R^k$ vadinsime bet kokià atviràjà lankiğkai jungià
aibæ $D \subset \R^k$.

Toliau visur $D \subset \R^k$ -- sritis.

Funkcijos $f : D \to \R$ \Def{gradientu} vadinama funkcija $\grad f : D \to
\R^k$:
\[ \grad f(x) := \left( \frac{\partial f}{\partial x_1}, \dots,
                        \frac{\partial f}{\partial x_k} \right),
              \qquad x \in D
\]
(kitas şymëjimas: $\grad f(x) = f'(x)$, jei vartojamas ne eilutinis, o
stulpelinis vektorius).

Funkcija $f: D \to \R^k$ vadinama \Def{potencialo lauku}, jei egzistuoja
funkcija $U: D \to \R$, su kuria $\grad U = f$ srityje $D$.  Tokiu atveju $U$
vadinama ,,jëgø lauko`` $f$ \Def{potencialu}.
\end{ap}

\begin{tr}[kreiviniai integralai ir integravimo kreivë]			% 7.13

\end{enumerate}

\begin{tr}[bûtina ir pakankama sàlyga, kad funkcija bûtø potencialo laukas]
\label{7.14}								% 7.14
Tarkime, $f \in C^1(D, \R^k)$.
\begin{enumerate}
\item Jei $f$ -- potencialo laukas, tai
\begin{equation*}
   \frac{\partial f_i}{\partial x_j} =
   \frac{\partial f_j}{\partial x_i},
   \qquad i, j = 1 \dots k, \text{ srityje $D$.}
   \tag*{($*$)}\label{eq:7.14.1}
\end{equation*}
\item Jei $D$ -- atviroji gardelë, ir joje patenkintos \ref{eq:7.14.1}
lygybës, tai $f$ -- potencialo laukas ir funkcija $U(x)$ yra jos potencialas:
\begin{align*}
  U(x) = U(x^0) &+ \int_{x^0_1}^{x_1} f_1(y, x^0_2, x^0_3, \dots, x^0_k)\,\dy \\
                &+ \int_{x^0_2}^{x_2} f_2(x_1, y, x^0_3, \dots, x^0_k)\,\dy \\
                &+ \ldots \\
                &+ \int_{x^0_k}^{x_k} f_k(x_1, \dots, x_{k-1}, y)\,\dy,
\qquad x \in D
\end{align*}
(tağkas $x^0 \in D$ ir $U(x^0)$ reikğmë pasirenkami laisvai).
\end{enumerate}
\end{tr}

class Document:
    """
    Abstraktus dokumentas.  Subklasës turëtø realizuoti metodus view,
    printIt bei edit.
    """

    def decorator(self, subtype):
        """Gràþina nurodyto tipo dekoratoriø.

        Jei dekoratoriø grandinëje yra daugiau nei vienas nurodyto tipo
        dekoratorius, graþina iðoriðkiausià.  Jei nurodyto tipo dekoratoriaus
        nëra, gràþina None.
        """
        return None

    def allDecorators(self, subtype):
        """Gràþina visus tipo dekoratorius sàraðe."""
        return []

    def unwrap(self, subtype, all):
        """Gràþina dokumentà be nurodyto tipo dekoratoriaus.

        Parametras all sako, ar reikia iðmesti visus nurodyto tipo dekoratorius
        (jei true), ar tik iðoriðkiausià (jei false).
        """
        return self

    def next(self):
        """Gràþina kità objektà dekoratoriø grandinëje.  Jei objektas
        nedekoruotas, gràþina None
        """
        return None

class Decorator(Document):
    """Abstraktus dekoratorius"""

    def __init__(self, next):
        """Konstruktorius."""
        self._next = next

    def __getattr__(self, attr):
        """Deleguojame priëjimà prie visø kitø atributø tikrajam objektui."""
        return getattr(self._next, attr)

    def decorator(self, subtype):
        if isinstance(self, subtype):
            return self
        else:
            return self._next.decorator(subtype)

    def allDecorators(self, subtype):
        if isinstance(self, subtype):
            result = [self]
        else:
            result = []
        return result + self._next.allDecorators(subtype)

    def unwrap(self, subtype, all):
        if isinstance(self, subtype):
            if all:
                return self._next.unwrap(subtype, all)
            else:
                return self._next
        else:
            return self._next.unwrap(subtype, all)

    def next(self):
        return self._next

class ReadOnly(Decorator):
    """Nemodifikuojamas dokumentas"""

    def edit(self):
        print "negalima redaguoti ðio dokumento!"

class Annotation(Decorator):
    """Anotacija skaitytojui"""

    def __init__(self, next, annotation):
        """Konstruktorius"""
        Decorator.__init__(self, next)
        self._annotation = annotation

    def view(self):
        self.next().view()
        print "  anotacija:", self._annotation

    def printIt(self):
        self.next().printIt()
        print "  anotacija:", self._annotation

    def annotation(self):
        """Gràþina anotacijà"""
        return self._annotation

    def changeAnnotation(self, annotation):
        """Pakeièia anotacijà"""
        self._annotation = annotation

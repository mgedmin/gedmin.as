 * @section problem Problema
 *
 * Projektuokime dokumentø tvarkymo sistemà.  Dokumentai gali bûti ávairiø
 * rûðiø (tekstas, paveiksliukai ir t.t.).  Dokumentus galime perþiûrinëti,
 * spausdinti, redaguoti.
 *
 * Taip pat norime keisti ar papildyti kai kuriø konkreèiø dokumentø elgesá.
 * Pavyzdþiui, uþdrausti kai kuriø dokumentø redagavimà, ar prikabinti prie
 * dokumentø anotacijas.
 *
 * Jei kiekvienam variantui kursime po atskirà klasæ, gausime kombinatoriná
 * sprogimà -- %TextDocument, ReadOnlyTextDocument, AnnotatedTextDocument,
 * AnnotatedReadOnlyTextDocument ir t.t.
 *
 * @section solution Sprendimas
 *
 * Pritaikysime Dekoratoriaus ðablonà.  Kiekvienà dokumento tipà atitinka
 * atskira klasë.  Visi dokumentai paveldi ið bendros astrakèios bazinës klasës
 * ir realizuoja tà patá interfeisà, nusakantá baziná dokumentø funkcionalumà.
 *
 * Papildomas savybes suteiks dekoratoriai.  Abstrakti dekoratoriaus klasë
 * paveldi ið bazinës dokumentø klasës ir realizuoja bendrà visiems
 * dekoratoriams funkcionalumà -- metodø delegavimà.
 *
 * Tradiciniame Dekoratoriaus ðablono variante dokumentai nieko neþino apie
 * dekoratorius, neþino net apie jø egzistavimà.  Taèiau tokiu atveju galime
 * lengvai nuimti tik iðoriná dekoratoriø, jei þinome jo tipà.  Taip pat galime
 * lengvai pasinaudoti tik iðorinio dekoratoriaus (jei þinome jo tipà)
 * pateikiamomis papildomomis funkcijomis, nes dekoratoriai deleguoja tik
 * dokumentø interfeiso funkcijas.  Norëtøsi ko nors lankstesnio.
 *
 * Turëdami koká nors objektà (neþinodami, ar jis dekoruotas, ar ne), mes
 * norëtumëme suþinoti (gauti nuorodà) á tam tikros konkreèios klasës
 * dekoratoriø (kad galëtumëme pasinaudoti jo teikiamomis papildomomis
 * funkcijomis).  Taip pat norëtumëme iðmesti ið dekoratoriø grandinës vienà
 * ar visus nurodyto tipo dekoratorius.
 *
 * Kad pasiektumëme papildomai iðkeltus tikslus turime kiek pakeisti tradiciná
 * Dekoratoriaus ðablonà ir pridëti kelias funkcijas á dokumentø bazinæ klasæ
 * (negalime jø pridëti á dekoratoriø bazinæ klasæ, nes norime ir nedekoruotus
 * objektus traktuoti vienodai).
 *
 * @section implementation Realizacija
 *
 * @image html classdiagram.png "Klasiø diagrama"
 * @image latex classdiagram.eps "Klasiø diagrama" width=12cm
 *
 * Klases galima sugrupuoti á tokias kategorijas:
 * - @ref interfaces "Interfeisai" (@ref Document, @ref Decorator)
 * - @ref documents  "Konkretûs dokumentai" (@ref TextDocument, @ref Image)
 * - @ref decorators "Konkretûs dekoratoriai" (@ref ReadOnly, @ref Annotation)
 *
 * Pagrindinëje programoje pateiktas @ref example "naudojimo pavyzdys".
 */

#include "decorator.h"

#include <iostream>

// Document

Document::~Document()
{}

// Decorator

Decorator::~Decorator()
{
    delete next_;
}

// TextDocument

void TextDocument::view()
{
    std::cout << "tekstinio dokumento perþiûra\n";
}

void TextDocument::print()
{
    std::cout << "tekstinio dokumento spausdinimas\n";
}

void TextDocument::edit()
{
    std::cout << "tekstinio dokumento redagavimas\n";
}

// Image

void Image::view()
{
    std::cout << "paveiksliuko perþiûra\n";
}

void Image::print()
{
    std::cout << "paveiksliuko spausdinimas\n";
}

void Image::edit()
{
    std::cout << "paveiksliuko redagavimas\n";
}

// ReadOnly

void ReadOnly::edit()
{
    std::cout << "negalima redaguoti ðio dokumento!\n";
}

// Annotation
void Annotation::view()
{
    Decorator::view();
    std::cout << "  anotacija: " << annotation() << '\n';
}

void Annotation::print()
{
    Decorator::print();
    std::cout << "  anotacija: " << annotation() << '\n';
}

/**
 * Pagrindinë programa, iliustruojanti dekoratoriø panaudojimà.  Jà ávykdþius
 * turëtø pasirodyti ðitoks tekstas:
@verbatim

  * Bazinis funkcionalumas
  tekstinio dokumento perþiûra
  tekstinio dokumento spausdinimas
  tekstinio dokumento redagavimas
  paveiksliuko perþiûra
  paveiksliuko spausdinimas
  paveiksliuko redagavimas

  * Bazinis funkcionalumas su dekoratoriais
  tekstinio dokumento perþiûra
    anotacija: ádomus dokumentas
  tekstinio dokumento spausdinimas
    anotacija: ádomus dokumentas
  negalima redaguoti ðio dokumento!
  paveiksliuko perþiûra
    anotacija: graþus paveiksliukas
    anotacija: o man nepatiko
  paveiksliuko spausdinimas
    anotacija: graþus paveiksliukas
    anotacija: o man nepatiko
  paveiksliuko redagavimas

  * Papildomas funkcionalumas: pakeiskime anotacijà
  tekstinio dokumento perþiûra
    anotacija: o gal ir nelabai ádomus

  * Pereikime per visus vieno tipo dekoratorius
  radau anotacijà: o man nepatiko
  radau anotacijà: graþus paveiksliukas

  * Nuimkime ReadOnly dekoratoriø
  tekstinio dokumento redagavimas

@endverbatim
 */
int main()
{
    // Sukuriame plikus objektus
    Document* txt = new TextDocument();
    Document* img = new Image();

    // Bazinis funkcionalumas
    cout << "* Bazinis funkcionalumas\n";
    txt->view();
    txt->print();
    txt->edit();
    img->view();
    img->print();
    img->edit();
    cout << '\n';

    // Prikabiname dekoratorius
    txt = new Annotation(txt, "ádomus dokumentas");
    txt = new ReadOnly(txt);

    img = new Annotation(img, "graþus paveiksliukas");
    img = new Annotation(img, "o man nepatiko");

    // Bazinis funkcionalumas su dekoratoriais
    cout << "* Bazinis funkcionalumas su dekoratoriais\n";
    txt->view();
    txt->print();
    txt->edit();
    img->view();
    img->print();
    img->edit();
    cout << '\n';

    // Papildomas funkcionalumas
    cout << "* Papildomas funkcionalumas: pakeiskime anotacijà\n";
    if (Annotation* a = txt->decorator<Annotation>()) {
	a->changeAnnotation("o gal ir nelabai ádomus");
	txt->view();
    };
    cout << '\n';

    // Parodykime visas anotacijas
    cout << "* Pereikime per visus vieno tipo dekoratorius\n";
    for (Document* tmp = img;
         Annotation* d = tmp->decorator<Annotation>();
         tmp = d->next())
    {
	std::cout << "radau anotacijà: " << d->annotation() << '\n';
    }
    cout << '\n';

    // Nuimkime dekoratoriø
    cout << "* Nuimkime ReadOnly dekoratoriø\n";
    txt = txt->unwrap<ReadOnly>(true);
    txt->edit();
    cout << '\n';

    // Panaikinkime objektus ir visus dekoratorius
    delete txt;
    delete img;
}

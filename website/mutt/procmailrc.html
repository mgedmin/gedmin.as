# Kurdamas ðá failiukà smarkiai rëmiausi Rejo Zenger procmailrc:
# http://www.mediaport.org/~sister/personal/procmailrc.

# $MAILDIR -- kalalogas, kuriame yra visi mano paðto aplankai
MAILDIR = $HOME/Mail
# $DEFAULT -- kur dëti laiðkà, jei nebus kitø nurodymø
DEFAULT = $MAILDIR/inbox
# (Pas mane tai symlinkas á /var/spool/mail/mg).

# Paranoja: kursim atsargines kiekvieno laiðko kopijas.  Kiekvienos dienos
# laiðkai bus atskirame faile.
DAY = `date +%Y-%m-%d`
BACKUP = $MAILDIR/backup/$DAY

# Kad galëtø atpaþinti laiðkø dublikatus, formail'as turi kaþkur ásiminti
# matytø Message-ID sàraðà.  Èia nurodoma jo vieta ir maksimalus dydis.
CACHE_SIZE = 8192
CACHE_FILE = $HOME/tmp/msgid.cache


# Visø pirma -- atsarginës kiekvieno laiðko kopijos.  Pastaba: tai smarkiai
# maþina laisvà vietà diske.
:0c:
$BACKUP

# Atpaþinsime pasikartojanèius laiðkus pagal Message-ID laukà.
:0Whc:msgid.lock
| formail -D $CACHE_SIZE $CACHE_FILE

# Jei atpaþinome -- pridëkime X-Note laukà
:0afhw
| formail -A "X-Note: Message-ID seen before recently."

# Sugrosime muzikëlæ atëjus asmeniniam laiðkui
:0hwic
| cat > /dev/null; /usr/bin/play $HOME/wavs/message.wav 2>/dev/null; true

# Jei jûs mëtytumëte áeinanèius laiðkus á skirtingas paðto dëþutes, èia
# reiktø priraðyti keletà ðtai ðitokiø taisykliø:
#
# :0:
# * ^TO_pasto@konferencijos.adresas
# pasto-dezutes-vardas
#

# Kas liko, keliauja á pagrindinæ paðto dëþutæ.
:0:
$DEFAULT
